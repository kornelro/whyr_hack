def row_to_comment(row):

    kids = []
    if row['kids'] != '{}' and type(row['kids']) == str:
        kids = list(map(int, row['kids'][1:-1].split(', ')))

    return Comment(
        by_id=row['by'],
        id=int(row['id']),
        parent_id=int(row['parent']),
        kids_ids=kids,
        text=row['text'],
        time=datetime.strptime(row['time'], '%Y-%m-%d %H:%M:%S'),
        type=row['type'],
        by=None,
        parent=None,
        kids=[]
    )

groups = []

for root in roots:

    group = []
    if root.by_id is not None:
        group.append(root.by_id)
    kids = root.kids

    while len(kids) > 0:
        kid = kids.pop()
        # group.append(kid)
        if kid.by_id is not None and kid.by_id not in group:
            group.append(kid.by_id)
        if kid.kids is not None:
            for k in kid.kids:   
                kids.append(k)

    groups.append(group)
